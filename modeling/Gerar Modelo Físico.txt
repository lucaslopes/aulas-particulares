CREATE TABLE USUARIO (
    id_usuario INTEGER PRIMARY KEY,
    nome CHARACTER,
    user_name CHARACTER,
    senha CHARACTER,
    e-mail CHARACTER,
    tel CHARACTER,
    professor BOOLEAN,
    ativo BOOLEAN
);

CREATE TABLE MATERIA (
    id_materia INTEGER PRIMARY KEY,
    desc CHARACTER
);

CREATE TABLE AGENDA (
    id_agenda INTEGER PRIMARY KEY,
    ativo BOOLEAN,
    dia_semana CHARACTER,
    inicio INTEGER,
    fim CHARACTER
);

CREATE TABLE AULA (
    id_aula INTEGER PRIMARY KEY,
    ativo BOOLEAN,
    FK_USUARIO_id_usuario INTEGER,
    FK_MATERIA_id_materia INTEGER,
    FK_AGENDA_id_agenda INTEGER
);

CREATE TABLE GRADE (
    id_grade INTEGER PRIMARY KEY,
    ativo BOOLEAN,
    FK_USUARIO_id_usuario INTEGER,
    FK_AULA_id_aula INTEGER
);
 
ALTER TABLE AULA ADD CONSTRAINT FK_AULA_1
    FOREIGN KEY (FK_MATERIA_id_materia)
    REFERENCES MATERIA (id_materia);
 
ALTER TABLE AULA ADD CONSTRAINT FK_AULA_2
    FOREIGN KEY (FK_USUARIO_id_usuario)
    REFERENCES USUARIO (id_usuario);
 
ALTER TABLE AULA ADD CONSTRAINT FK_AULA_3
    FOREIGN KEY (FK_AGENDA_id_agenda)
    REFERENCES AGENDA (id_agenda);
 
ALTER TABLE GRADE ADD CONSTRAINT FK_GRADE_1
    FOREIGN KEY (FK_USUARIO_id_usuario)
    REFERENCES USUARIO (id_usuario);
 
ALTER TABLE GRADE ADD CONSTRAINT FK_GRADE_2
    FOREIGN KEY (FK_AULA_id_aula)
    REFERENCES AULA (id_aula);
